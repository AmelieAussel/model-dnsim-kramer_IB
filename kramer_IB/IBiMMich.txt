gM = [0.75]
E_M = [-95]
c_MaM = [.625]
c_MbM = [2.75]
IC = [0]
IC_noise = [0.01]
Qs = 3.209^2
halfMM = -50

aM(X) = c_MaM.*(0.0001*Qs*(X-halfMM)) ./ (1-exp(-(X-halfMM)/9))
bM(X) = c_MbM.*(-0.0001*Qs*(X-halfMM)) ./ (1-exp((X-halfMM)/9))
IM(X,mM) = gM.*mM.*(X-E_M)
infmM(X) = aM(X)./(aM(X)+bM(X))
taumM(X) = 1./(aM(X)+bM(X))

mM' = (aM(V).*(1-mM)-bM(V).*mM);
mM(0) = IC+IC_noise.*rand(Npop,1)

% monitor functions % always record the synaptic current
 
@current += -IM(V,mM)
