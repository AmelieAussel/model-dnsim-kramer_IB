ac = 0;
freq = 0;
phase = 0;
onsetRAN = 0;
T=2000; 	% ms, duration
tauRAN=4; 	% ms, exponential decay time constant
T2=(0:0.01:T)';

s=getPoissonGating(baseline,lambda,ac,freq,phase,onsetRAN,T,tauRAN,T2,Npop);
Iext(X)=gRAN*s(k,:).*(X-ERAN)
